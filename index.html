// --- LOGICA DI CONTROLLO DIRETTO OTTIMIZZATA --- let currentFrame = 0; let targetFrame = 0; // REATTIVITÀ: // 0.1 = Molto pesante/lento (effetto gomma) // 0.4 = Molto leggero (quasi istantaneo) // 0.8 = Secco (si ferma nell'istante in cui ti fermi tu) const smoothness = 0.45; // SOGLIA DI TOLERANZA: // Evita che il processore faccia calcoli inutili quando sei quasi arrivato al frame const epsilon = 0.01; // --- 3. LOGICA DI POSIZIONAMENTO --- window.addEventListener("scroll", () => { const scrollTop = window.scrollY; const maxScroll = document.documentElement.scrollHeight - window.innerHeight; // Calcolo ultra-preciso della posizione const scrollFraction = Math.max(0, Math.min(1, scrollTop / maxScroll)); if (scrollTop > 10) { isScrolling = true; targetFrame = scrollFraction * (totalCartella2 - 1); } else { isScrolling = false; } }); // --- 4. RENDERING AD ALTE PRESTAZIONI --- function animate() { if (isScrolling) { const diff = targetFrame - currentFrame; // Se la differenza è minima, saltiamo direttamente al target // così non senti quell'effetto di "trascinamento lento" alla fine if (Math.abs(diff) < epsilon) { currentFrame = targetFrame; } else { currentFrame += diff * smoothness; } const frameToDraw = Math.round(currentFrame); context.clearRect(0, 0, canvas.width, canvas.height); if (immaginiScroll[frameToDraw]) { context.drawImage(immaginiScroll[frameToDraw], 0, 0); } } requestAnimationFrame(animate); } animate();
